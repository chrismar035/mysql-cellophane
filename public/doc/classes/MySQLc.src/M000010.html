  def delete(*vals)
    set_attributes vals

    @query = ""
    if @id_field != ""
      @query += "WHERE #{esc @id_field} = '#{esc @id}'"
    end

    if @extra_cond != "" && !@extra_cond.nil?
      @query += ((@query == "") ? " WHERE " : " AND ") + @extra_cond
    end

    @query = "DELETE FROM #{@table} #{@query}"
    execute
  end
