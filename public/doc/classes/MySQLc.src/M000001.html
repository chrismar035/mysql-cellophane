  def initialize(options = {})
    log = File.new('sql_log.log', 'a')
    log.puts "Created New MySQLc instance"
    log.close

    options[:host]      ||= 'localhost'
    options[:user]      ||= 'root'
    options[:password]  ||= 'secure'
    options[:database]  ||= 'test'
    options[:port]      ||= nil
    options[:socket]    ||= nil
    options[:flag]      ||= nil

    @options = {}
    @options.merge!(options)

    @field_list = "*"
    @id_field   = 1
    @id         = 1
    @cache      = {}
  end
